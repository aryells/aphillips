54      @Test
55      public void callsAllProviders() {
56          ClassLoader classpathResourceLoader = createNiceMock(ClassLoader.class);
57          ServiceImplementationProvider provider1 = createMock(ServiceImplementationProvider.class);
58          expect(provider1.findServiceImplementations(Agent.class, classpathResourceLoader))
59          .andReturn(newHashSet(JamesBond.class.getName(), JackGiddings.class.getName()));
60          ServiceImplementationProvider provider2 = createMock(ServiceImplementationProvider.class);
61          expect(provider2.findServiceImplementations(Agent.class, classpathResourceLoader))
62          .andReturn(newHashSet(JamesBond.class.getName(), StuartThomas.class.getName()));
63          replay(provider1, provider2);
64          
65          multiSpi = new MultiSpi(newHashSet(provider1, provider2));
66          assertEquals(newHashSet(JamesBond.class.getName(), JackGiddings.class.getName(), 
67                      StuartThomas.class.getName()),
68                  multiSpi.findImplementationNames(Agent.class, classpathResourceLoader));
69          verify(provider1, provider2);
70      }

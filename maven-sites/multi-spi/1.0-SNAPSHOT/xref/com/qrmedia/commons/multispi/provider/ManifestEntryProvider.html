24  import static com.google.common.collect.Iterables.transform;
25  import static com.google.common.collect.Maps.filterValues;

27  import java.io.IOException;
28  import java.net.URL;
29  import java.util.Set;
30  import java.util.jar.Attributes;
31  import java.util.jar.Manifest;

33  import javax.annotation.Nonnull;

35  import org.apache.commons.lang.StringUtils;
36  
37  import com.google.common.base.Function;
38  import com.google.common.base.Predicate;
39  import com.google.common.collect.ImmutableSet;

41  /***
42   * A {@link ServiceImplementationProvider} that returns implementations listed in 
43   * a <a href="http://download.oracle.com/javase/6/docs/technotes/guides/jar/jar.html#Service%20Provider">Java Manifest file</a>,
44   * based on properties of the implementation's manifest entry.   
45   * 
46   * @author aphillips
47   * @since 4 Dec 2010
48   *
49   */
50  abstract class ManifestEntryProvider extends ClasspathResourcesReadingProvider<Manifest> {
51      private static final String MANIFEST_FILE_NAME = "META-INF/MANIFEST.MF";
52      private static final String CLASS_FILE_SUFFIX = ".class";
53      private static final char RESOURCE_PATH_SEPARATOR = '/';
54      private static final char PACKAGE_SEPARATOR = '.';
55  
56      protected ManifestEntryProvider() {
57          super(constant(MANIFEST_FILE_NAME), new IoFunction<URL, Manifest>() {
58                  public Manifest apply(URL item) throws IOException {
59                      return new Manifest(item.openStream());
60                  }
61              });
62      }
63      
64      protected Set<String> processResource(Manifest manifest, final Class<?> serviceClass) {
65          return ImmutableSet.copyOf(transform(
66                  filterValues(manifest.getEntries(), new Predicate<Attributes>() {
67                          public boolean apply(Attributes input) {
68                              return implementationManifestEntry(serviceClass, input);
69                          }
70                      })
71                  .keySet(), 
72                  new ResourcePathsToClassNames()));
73      }
74      
75      /***
76       * @param serviceClass the class of the service requested
77       * @param entryAttributes the attributes of a Java manifest entry
78       * @return {@code true} iff the attributes indicate that the manifest entry is
79       *         an implementation of the service class
80       */
81      protected abstract boolean implementationManifestEntry(@Nonnull Class<?> serviceClass, 
82              @Nonnull Attributes entryAttributes);
83  
84      private static class ResourcePathsToClassNames implements Function<String, String> {
85          public String apply(String from) {
86              return StringUtils.removeEnd(from, CLASS_FILE_SUFFIX)
87                     .replace(RESOURCE_PATH_SEPARATOR, PACKAGE_SEPARATOR);
88          }
89      }
90  }

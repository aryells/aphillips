43  /***
44   * One-stop shop for users of MultiSpi. Instances should be created via dependency
45   * injection or using the {@link MultiSpiBuilder}.
46   * 
47   * @author aphillips
48   * @since 4 Dec 2010
49   *
50   */
51  @ThreadSafe
52  public final class MultiSpi {
53      private final Set<ServiceImplementationProvider> providers = newHashSet();
54      
55      @Inject
56      public MultiSpi(@Nonnull Set<ServiceImplementationProvider> providers) {
57          this.providers.addAll(providers);
58      }
59      
60      /***
61       * Finds the names of classes implementing the requested service, as determined
62       * by the available providers.
63       * <p>
64       * Note that individual providers may or may <u>not</u> guarantee that the classes
65       * returned <em>actually</em> implement the interface!
66       * 
67       * @param serviceClass the class of the service
68       * @return a set of names of classes designated as implementing the requested service
69       */
70      public @Nonnull Set<String> findImplementationNames(@Nonnull final Class<?> serviceClass) {
71          return newHashSet(concat(transform(providers, new Function<ServiceImplementationProvider, Set<String>>() {
72                  public Set<String> apply(ServiceImplementationProvider from) {
73                      return from.findServiceImplementations(serviceClass);
74                  }
75              })));
76      }
77      
78      /***
79       * Shorthand for {@link #findImplementations(Class, MultiSpi.class.getClassLoader())}.
80       * 
81       * @param <S> the type of the service
82       * @param serviceClass the class of the service
83       * @return a set of classes of implementations of the service
84       * @throws ClassCastException if any of the classes are not actually implementations
85       *                            of the requested service
86       * @throws ClassNotFoundException if any of the classes cannot be loaded    
87       * @see #findImplementations(Class, ClassLoader)
88       */
89      public @Nonnull <S> Set<Class<? extends S>> findImplementations(
90              @Nonnull Class<S> serviceClass) throws ClassCastException, ClassNotFoundException {
91          return findImplementations(serviceClass, MultiSpi.class.getClassLoader());
92      }
93      
94      /***
95       * Finds the classes implementing the requested service, as determined by the 
96       * available providers.
97       * <p>
98       * This method will check whether the classes found do, in fact, implement the
99       * requested service, and a {@code ClassNotFoundException} will be thrown if any
100      * of the classes does not.
101      * <p>
102      * If more lenient behaviour is desired (e.g. skipping any classes that cannot be
103      * found, or do not actually implement the interface), use 
104      * {@link #findImplementationNames(Class)} instead.
105      * 
106      * @param <S> the type of the service
107      * @param serviceClass the class of the service
108      * @param implementationLoader the class loader to be used to attempt to load the classed
109      * @return a set of classes of implementations of the service
110      * @throws ClassCastException if any of the classes are not actually implementations
111      *                            of the requested service
112      * @throws ClassNotFoundException if any of the classes cannot be loaded                           
113      * @see #findImplementations(Class)
114      */
115     @SuppressWarnings("unchecked")
116     public @Nonnull <S> Set<Class<? extends S>> findImplementations(
117             @Nonnull Class<S> serviceClass, @Nonnull final ClassLoader implementationLoader) 
118             throws ClassCastException, ClassNotFoundException {
119         Set<Class<? extends S>> implementations = newHashSet();
120         // can't do this using Iterables.transform because we want to throw the exception
121         for (String name : findImplementationNames(serviceClass)) {
122             implementations.add((Class<? extends S>) implementationLoader.loadClass(name));
123         }
124         
125         if (any(implementations, not(assignableFrom(serviceClass)))) {
126             throw new ClassCastException(format("Supposed implementations '%s' does/do not implement or extend service class '%s'",
127                     filter(implementations, not(assignableFrom(serviceClass))), serviceClass.getName()));
128         }
129         return implementations;
130     }
131     
132     private static AssignableFromPredicate assignableFrom(Class<?> superclass) {
133         return new AssignableFromPredicate(superclass);
134     }
135 
136     private static class AssignableFromPredicate implements Predicate<Class<?>> {
137         private final Class<?> superclass;
138         
139         public AssignableFromPredicate(Class<?> superclass) {
140             this.superclass = superclass;
141         }
142 
143         public boolean apply(Class<?> input) {
144             return superclass.isAssignableFrom(input);
145         }
146     }

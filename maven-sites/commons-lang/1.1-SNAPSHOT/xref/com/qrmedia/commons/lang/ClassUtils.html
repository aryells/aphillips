52      /***
53       * Returns <u>one</u> of the possible chains of superclasses and/or interfaces joining the
54       * specified class to the given superclass, as returned by {@link #getSuperclassChains(Class, Class)}.
55       * <i>Which</i> of the possible chains will be returned is not defined.
56       * <p>
57       * If <code>superclass</code> is <i>not</i> a superclass or -interface of <code>class</code>,
58       * the method returns <code>null</code>. This may happen (in spite of the signature) if the 
59       * method is called with non-generic arguments.
60       *
61       * @param <S>       the type of the superclass at the &quot;end&quot; of the chain
62       * @param clazz     the class at the &quot;start&quot; of the superclass chain
63       * @param superclass        the class at the &quot;end&quot; of the superclass chain
64       * @return <u>one</u> superclass chain linking <code>class</code> to <code>superclass</code>,
65       *         where successive elements of the list are immediate superclasses or -interfaces. If
66       *         <code>class</code> is not a subclass of <code>superclass</code>, returns <code>null</code>.
67       * @throws IllegalArgumentException if either argument is null    
68       * @see #getSuperclassChains(Class, Class)    
69       */
70      public static <S> List<Class<? extends S>> getSuperclassChain(Class<? extends S> clazz, 
71              Class<S> superclass) {
72          Set<List<Class<? extends S>>> superclassChains = getSuperclassChainsInternal(clazz, superclass, true);
73          return (superclassChains.isEmpty() ? null : superclassChains.iterator().next());
74      }

76      /***
77       * Returns the chain of superclass and/or interfaces from the specified class to the given
78       * superclass. Either parameter may be an interface.
79       * <p>
80       * Each list in the resulting set contains immediate superclass elements in order, i.e. for
81       * classes
82       * 
83       * <pre>
84       * class Foo {}
85       * class Bar extends Foo {}
86       * class Baz extends Bar {}
87       * </pre>
88       * 
89       * <code>getSuperclassChains(Baz.class, Foo.class)</code> will return one list, <code>[Baz.class,
90       * Bar.class, Foo.class]</code>.
91       * <p>
92       * If both parameters are classes, there can only be one possible chain. However, if the superclass
93       * is an interface, there may be multiple possible inheritance chains. For instance, for
94       * 
95       * <pre>
96       * interface Foo {}
97       * interface Bar1 extends Foo {}
98       * interface Bar2 extends Foo {}
99       * interface Baz extends Bar1, Bar2 {}
100      * </pre>
101      * 
102      * both <code>[Baz.class, Bar1.class, Foo.class]</code> and <code>[Baz.class, Bar2.class, Foo.class]</code>
103      * are valid inheritance chains, and the method returns both.
104      * <p>
105      * If <code>superclass</code> is <i>not</i> a superclass or -interface of <code>class</code>,
106      * the method returns an empty set. This may happen (in spite of the signature) if the 
107      * method is called with non-generic arguments.
108      *
109      * @param <S>       the type of the superclass at the &quot;end&quot; of the chain 
110      * @param clazz     the class at the &quot;start&quot; of the superclass chain
111      * @param superclass        the class at the &quot;end&quot; of the superclass chain
112      * @return all possible superclass chains linking <code>class</code> to <code>superclass</code>,
113      *         where successive elements of the list are immediate superclasses or -interfaces. If
114      *         <code>class</code> is not a subclass of <code>superclass</code>, returns an empty set.
115      * @throws IllegalArgumentException if either argument is null  
116      * @see #getSuperclassChain(Class, Class)      
117      */
118     public static <S> Set<List<Class<? extends S>>> getSuperclassChains(Class<? extends S> clazz, Class<S> superclass) {
119         return getSuperclassChainsInternal(clazz, superclass, false);
120     }

122     private static <S> Set<List<Class<? extends S>>> getSuperclassChainsInternal(Class<? extends S> clazz, 
123             Class<S> superclass, boolean oneChainSufficient) {
124         checkNotNull("'clazz' and 'superclass' may not be non-null", clazz, superclass);
125         
126         if (!superclass.isAssignableFrom(clazz)) {
127             return Collections.emptySet();
128         }
129         
130         // interfaces only need to be considered if the superclass is an interface
131         return ClassUtils.getSuperclassSubchains(clazz, superclass, 
132                 oneChainSufficient, superclass.isInterface());
133     }

135     // recursive method: gets the subchains from the given class to the target class
136     @SuppressWarnings("unchecked")
137     private static <S> Set<List<Class<? extends S>>> getSuperclassSubchains(
138             Class<? extends S> subclass, Class<S> superclass, boolean oneChainSufficient, 
139             boolean considerInterfaces) {
140         
141         // base case: the subclass *is* the target class
142         if (subclass.equals(superclass)) {
143             
144             // since the list will be built from the *head*, a linked list is a good choice
145             List<Class<? extends S>> subchain = new LinkedList<Class<? extends S>>();
146             subchain.add(subclass);
147             return Collections.singleton(subchain);
148         }
149         
150         // recursive case: get all superclasses and, if required, interfaces and recurse
151         Set<Class<? extends S>> supertypes = new HashSet<Class<? extends S>>();
152         
153         Class<? extends S> immediateSuperclass = (Class<? extends S>) subclass.getSuperclass();
154         
155         // interfaces and Object don't have a superclass
156         if (immediateSuperclass != null) {
157             supertypes.add(immediateSuperclass);
158         }
159         
160         if (considerInterfaces) {
161             supertypes.addAll(Arrays.asList((Class<? extends S>[]) subclass.getInterfaces()));
162         }
163         
164         Set<List<Class<? extends S>>> subchains = new HashSet<List<Class<? extends S>>>();
165         
166         for (Class<? extends S> supertype : supertypes) {
167             
168             // the compiler complains if the type parameters to getSuperclassSubchains aren't specified 
169             Set<List<Class<? extends S>>> subchainsFromSupertype = 
170                 ClassUtils.getSuperclassSubchains(supertype, superclass, 
171                         oneChainSufficient, considerInterfaces);
172             
173             // each chain from the supertype results in a chain [current, subchain-from-super]
174             if (!subchainsFromSupertype.isEmpty()) {
175                 
176                 if (oneChainSufficient) {
177                     ClassUtils.<S>addSubchain(subchains, subclass, 
178                             subchainsFromSupertype.iterator().next());
179                     return subchains;
180                 } else {
181                     
182                     for (List<Class<? extends S>> subchainFromSupertype : subchainsFromSupertype) {
183                         ClassUtils.<S>addSubchain(subchains, subclass, subchainFromSupertype);
184                     }
185                     
186                 }
187                 
188             }
189             
190         }
191         
192         return subchains;
193     }

195     // adds the class to the beginning of the subchain and stores this extended subchain
196     private static <T> void addSubchain(Set<List<Class<? extends T>>> subchains, 
197             Class<? extends T> clazz, List<Class<? extends T>> subchainFromSupertype) {
198         subchainFromSupertype.add(0, clazz);
199         subchains.add(subchainFromSupertype);
200     }

202     /***
203      * Determines if <i>any</i> of the given superclasses is a superclass or -interface of the
204      * specified class. See {@link Class#isAssignableFrom(Class)}.
205      * 
206      * @param superclasses      the superclasses and -interfaces to be tested against
207      * @param clazz     the class to be to be checked
208      * @return  <code>true</code> iff <i>any</i> of the given classes is a superclass or -interface
209      *          of the specified class
210      * @throws IllegalArgumentException if either of the arguments is null
211      */
212     public static boolean isAnyAssignableFrom(Collection<Class<?>> superclasses,
213             Class<?> clazz) {
214         checkNotNull("All arguments must be non-null", superclasses, clazz);
215 
216         for (Class<?> superclass : superclasses) {
217             
218             if (superclass.isAssignableFrom(clazz)) {
219                 return true;
220             }
221             
222         }
223         
224         return false;
225     }

229      * with <em>any</em> of the specified {@code Class Classes}. class. 
230      * See {@link Class#isInstance(Object)}.
231      * 
232      * @param superclasses      the superclasses and -interfaces to be tested against
233      * @param instance     the object to be to be checked
234      * @return  <code>true</code> iff the object is an instance of <i>any</i> of the

238     public static boolean isInstance(Collection<Class<?>> superclasses,
239             Object instance) {
240         checkNotNull("Superclasses non-null", superclasses);
241 
242         for (Class<?> superclass : superclasses) {
243             
244             if (superclass.isInstance(instance)) {
245                 return true;
246             }
247

267     /***
268      * Collects all fields declared in the given class <u>and</u> inherited from 
269      * parents <strong>up to and including the given parent class</strong>.
270      * 
271      * @param <T> the type of the class to parse
272      * @param clazz the class to parse
273      * @param superclass the superclass of the class to parse at which traversal should be
274      *                   stopped
275      * @return  a list of all fields declared in the class or its parents up to and including
276      *          the given parent class, in the order determined by successive 
277      *          {@link Class#getDeclaredFields()} calls
278      * @see #getAllDeclaredFields(Class)         
279      */
280     public static <T> List<Field> getAllDeclaredFields(Class<T> clazz, 
281             Class<? super T> superclass) {
282         final List<Field> fields = new ArrayList<Field>();
283         
284         for (Class<?> immediateSuperclass : getSuperclassChain(clazz, superclass)) {
285             fields.addAll(Arrays.asList(immediateSuperclass.getDeclaredFields()));
286         }
287         
288         return fields;
289     }

329             Class<? extends Annotation> annotationType) {
330         checkNotNull("'clazz' must be non-null", clazz);
331         
332         // perhaps this case should throw an exception, but an empty list also seems sensible 
333         if (annotationType == null) {

336

350     /***
351      * Retrieves the type arguments of a class when regarded as an subclass of the
352      * given typed superclass or interface. The order of the runtime type classes matches the order
353      * of the type variables in the declaration of the typed superclass or interface.
354      * <p>
355      * For example, for the classes
356      * 
357      * <pre>
358      * class Foo&lt;U, V&gt; {}
359      * class Bar&lt;W&gt; extends Foo&lt;String, W&gt; {}
360      * class Baz extends Bar&lt;Long&gt;
361      * </pre>
362      * 
363      * and a <code>typedClass</code> argument of <code>Baz.class</code>, the method should return
364      * <p>
365      * <ul>
366      * <li><code>[String, Long]</code> for a <code>typedSuperclass</code> argument of <code>Foo.class</code>,
367      *     and
368      * <li><code>[Long]</code> if <code>typedSuperclass</code> is <code>Bar.class</code>.
369      * </ul>
370      * For type parameters that cannot be determined, <code>null</code> is returned.
371      * <p>
372      * <b>Note:</b> It is <u>not</u> possible to retrieve type information that is not available
373      * in the (super)class hierarchy at <u>compile</u>-time. Calling 
374      * <code>getActualTypeArguments(new ArrayList&lt;String&gt;().getClass(), List.class)</code> will, 
375      * for instance, return <code>[null]</code> because the specification of the actual type 
376      * (<code>String</code>, in this example) did not take place either in the superclass {@link AbstractList} 
377      * or the interface {@link List}.
378      * <p>
379      * If {@code superclass} is <em>not</em> a superclass or -interface of {@code class},
380      * the method returns {@code null}. This may happen (in spite of the signature) if the 
381      * method is called with non-generic arguments.
382      * 
383      * @param <S>       the type of the object
384      * @param typedClass the class for which type information is required
385      * @param typedSuperclass the typed class or interface of which the object is to be regarded a 
386      *                        subclass
387      * @return  the type arguments for the given class when regarded as a subclass of the
388      *          given typed superclass, in the order defined in the superclass. If
389      *          {@code class} is not a subclass of {@code superclass}, returns {@code null}.
390      * @throws IllegalArgumentException if <code>typedSuperclass</code> or <code>typedClass</code> 
391      *                                  is <code>null</code>         
392      */
393     public static <S> List<Class<?>> getActualTypeArguments(Class<? extends S> typedClass,
394             Class<S> typedSuperclass) {
395         checkNotNull("All arguments must be non-null", typedSuperclass, typedClass);
396         
397         /*
398          * The type signature should ensure that the class really *is* an subclass of 
399          * typedSuperclass, but this can be circumvented by using "generic-less" arguements.
400          */
401         if (!typedSuperclass.isAssignableFrom(typedClass)) {
402             return null;
403         }
404 
405         TypeVariable<?>[] typedClassTypeParams = typedSuperclass.getTypeParameters();
406         
407         // if the class has no parameters, return
408         if (typedClassTypeParams.length == 0) {
409             return new ArrayList<Class<?>>(0);
410         }
411         
412         /*
413          * It would be nice if the parent class simply "aggregated" all the type variable
414          * assignments that happen in subclasses. In other words, it would be nice if, in the
415          * example in the JavaDoc, new Baz().getClass().getSuperclass().getGenericSuperclass()
416          * would return [String, Long] as actual type arguments.
417          * Unfortunately, though, it returns [String, W], because the assignment of Long to W
418          * isn't accessible to Bar. W's value is available from new Baz().getClass().getGenericSuperclass(),
419          * and must be "remembered" as we traverse the object hierarchy.
420          * Note, though, that the "variable substitution" of W (the variable used in Bar) for V (the
421          * equivalent variable in Foo) *is* propagated, but only to the immediate parent!
422          */
423         Map<TypeVariable<?>, Class<?>> typeAssignments = 
424             new HashMap<TypeVariable<?>, Class<?>>(typedClassTypeParams.length);
425         
426         /*
427          * Get one possible path from the typed class to the typed superclass. For classes, there
428          * is only one (the superclass chain), but for interfaces there may be multiple. We only
429          * need one, however (and it doesn't matter which one) since the compiler does not allow
430          * inheritance chains with conflicting generic type information.
431          */
432         List<Class<? extends S>> superclassChain = getSuperclassChain(typedClass, typedSuperclass);
433         
434         assert (superclassChain != null) : Arrays.<Class<?>>asList(typedSuperclass, typedClass);
435         
436         /*
437          * The list is ordered so that successive elements are immediate superclasses. The iteration
438          * stops with the class whose *superclass* is the last element, because type information
439          * is collected from the superclass.
440          */
441         for (int i = 0; i < superclassChain.size() - 1; i++) {
442             collectAssignments(superclassChain.get(i), superclassChain.get(i + 1), typeAssignments);
443         }
444         
445         // will contain null for entries for which no class could be resolved
446         return getActualAssignments(typedClassTypeParams, typeAssignments);
447     }

449     private static void collectAssignments(Class<?> clazz, Class<?> supertype,
450             Map<TypeVariable<?>, Class<?>> typeAssignments) {
451         TypeVariable<?>[] typeParameters = supertype.getTypeParameters();
452         
453         // the superclass is not necessarily a generic class
454         if (typeParameters.length == 0) {
455             return;
456         }
457         
458         Type[] actualTypeAttributes = getActualTypeAttributes(clazz, supertype);
459 
460         assert (typeParameters.length == actualTypeAttributes.length) 
461         : Arrays.asList(typeParameters, typeAssignments);
462         
463         // matches up type parameters with their actual assignments, assuming the order is the same!
464         for (int i = 0; i < actualTypeAttributes.length; i++) {
465             Type type = actualTypeAttributes[i];
466             
467             /*
468              * type will be a Class or ParameterizedType if the actual type is known, 
469              * and a TypeVariable if not.
470              */
471             if (type instanceof Class) {
472                 typeAssignments.put(typeParameters[i], (Class<?>) type);
473             } else if (type instanceof ParameterizedType) {
474                 assert (((ParameterizedType) type).getRawType() instanceof Class) : type;
475                 typeAssignments.put(typeParameters[i], 
476                         (Class<?>) ((ParameterizedType) type).getRawType());
477             } else {
478                 assert (type instanceof TypeVariable<?>) : type;
479                 
480                 /*
481                  * The actual type arguments consist of classes and type variables from the
482                  * immediate child class. So if the type assignment mapping is updated to
483                  * contain the mapping of all type variables of the *current* class to
484                  * their classes, then these can be used in the *next* iteration to resolve
485                  * any variable "left over" from this round.
486                  * 
487                  * Any variables that cannot be resolved in this round are not resolvable, otherwise
488                  * the would have been resolved in the previous round.
489                  */
490                 if (typeAssignments.containsKey(type)) {
491                     typeAssignments.put(typeParameters[i], typeAssignments.get(type));
492                 }
493                 
494             }
495             
496         }
497 
498     }

500     private static Type[] getActualTypeAttributes(Class<?> clazz, Class<?> supertype) {
501         /*
502          * The superclass is not necessarily a ParameterizedType even if it has type
503          * parameters! This happens if a user fails to specify type parameters for a
504          * class and ignores the warning, e.g.
505          * 
506          * class MyList extends ArrayList
507          * 
508          * In this case, the superclass ArrayList.class has one type parameter, but
509          * MyList.class.getGenericSuperclass() returns a simple type object!
510          * 
511          * In this case, no type assignments take place, so the actual arguments are
512          * simply the type parameters.
513          */    
514         Type genericSupertype = tryGetGenericSupertype(clazz, supertype);
515         return ((genericSupertype instanceof ParameterizedType)
516                 ? ((ParameterizedType) genericSupertype).getActualTypeArguments()
517                 : supertype.getTypeParameters());        
518     }

520     private static Type tryGetGenericSupertype(Class<?> clazz, Class<?> supertype) {
521         
522         if (!supertype.isInterface()) {
523             return clazz.getGenericSuperclass();
524         } else {
525             Type[] genericInterfaces = clazz.getGenericInterfaces();
526             
527             for (int i = 0; i < genericInterfaces.length; i++) {
528                 Type interfaceType = genericInterfaces[i]; 
529                 
530                 // there is no guarantee that *all* the interfaces are generic
531                 if ((interfaceType instanceof ParameterizedType) 
532                         && (((ParameterizedType) interfaceType).getRawType().equals(supertype))) {
533                     return interfaceType;
534                 } else {
535                     assert (interfaceType instanceof Class) : interfaceType;
536                     
537                     if (interfaceType.equals(supertype)) {
538                         return interfaceType;
539                     }   
540                     
541                 }
542                 
543             }
544             
545         }
546         
547         throw new AssertionError("Unable to find generic superclass information for class '"
548                                  + clazz + "' and superclass/-interface '" + supertype + "'");
549     }

551     private static List<Class<?>> getActualAssignments(
552             TypeVariable<?>[] typedClassTypeParams,
553             Map<TypeVariable<?>, Class<?>> typeAssignments) {
554         int numTypedClassTypeParams = typedClassTypeParams.length;
555         List<Class<?>> actualAssignments = 
556             new ArrayList<Class<?>>(numTypedClassTypeParams);
557         
558         // for entries that could not be resolved, null should be returned
559         for (int i = 0; i < numTypedClassTypeParams; i++) {
560             actualAssignments.add(typeAssignments.get(typedClassTypeParams[i]));
561         }
562         
563         return actualAssignments;
564     }

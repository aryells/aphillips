19  package com.qrmedia.commons.collections;
20  
21  import org.apache.commons.lang.builder.EqualsBuilder;
22  import org.apache.commons.lang.builder.HashCodeBuilder;
23  
24  /***
25   * A <code>Pair</code> whose equivalence respects the order of the two items.
26   * 
27   * @param <U> the type of the first object
28   * @param <V> the type of the second object
29   * @author anph
30   * @since 29 Aug 2008
31   *
32   */
33  public class OrderedPair<U, V> extends Pair<U, V> {
34  
35      /***
36       * Creates a ordered pair consisting of two objects.
37       * 
38       * @param firstObject    the first object of the pair
39       * @param secondObject  the second object of the pair
40       */
41      public OrderedPair(U firstObject, V secondObject) {
42          super(firstObject, secondObject);
43      }
44      
45      /* (non-Javadoc)
46       * @see com.tomtom.commons.collections.Pair#equals(java.lang.Object)
47       */
48      @Override
49      public boolean equals(Object obj) {
50          
51          if (this == obj) {
52              return true;
53          }
54  
55          if (!(obj instanceof Pair)) {
56              return false;
57          }
58  
59          Pair<?, ?> other = (Pair<?, ?>) obj;
60          
61          return new EqualsBuilder().append(firstObject, other.firstObject)
62                 .append(secondObject, other.secondObject).isEquals();
63      }
64  
65      /* (non-Javadoc)
66       * @see com.tomtom.commons.collections.Pair#hashCode()
67       */
68      @Override
69      public int hashCode() {
70          
71          // HashCodeBuilder is order-sensitive
72          return new HashCodeBuilder().append(firstObject).append(secondObject).toHashCode();
73      }
74  
75      /* (non-Javadoc)
76       * @see com.tomtom.commons.collections.Pair#toString()
77       */
78      @Override
79      public String toString() {
80          
81          // use "[" rather than the parent's "{" to hopefully indicate ordering
82          return "[" + getPairStringRepresentation() + "]";
83      }    
84  
85  }

34  /***
35   * Unit tests for the {@link SetUtils}.
36   * 
37   * @author anph
38   * @since 29 Aug 2008
39   */
40  @RunWith(value = Parameterized.class)
41  public class SetUtilsTest {
42      private Object[] objects;
43      private Set<Object> expectedSet;
44      
45      @Parameters
46      public static Collection<Object[]> data() {
47          List<Object[]> data = new ArrayList<Object[]>();
48          
49          // nulls
50          data.add(new Object[] { new Object[] { null } , asSet((Object) null) });
51          data.add(new Object[] { new Object[] { null, null } , asSet((Object) null) });
52          
53          // objects (some duplicates)
54          data.add(new Object[] { new Object[] { Long.valueOf(5) } , asSet(Long.valueOf(5)) });
55          data.add(new Object[] { new Object[] { Long.valueOf(5), Long.valueOf(6) } , 
56                                  asSet(Long.valueOf(5), Long.valueOf(6)) });
57          data.add(new Object[] { new Object[] { Long.valueOf(5), Long.valueOf(5) } , 
58                                  asSet(Long.valueOf(5)) });        
59          
60          // mixed objects
61          data.add(new Object[] { new Object[] { Long.valueOf(5), Integer.valueOf(6) } , 
62                                  asSet(Long.valueOf(5), Integer.valueOf(6)) });        
63          data.add(new Object[] { new Object[] { Long.valueOf(5), Integer.valueOf(5) } , 
64                                  asSet(Long.valueOf(5), Integer.valueOf(5)) });
65          
66          return data;
67      }
68      
69      private static Set<Object> asSet(Object... objs) {
70          Set<Object> set = new HashSet<Object>(objs.length);
71          
72          for (Object obj : objs) {
73              set.add(obj);
74          }
75          
76          return set;
77      }
78      
79      // called for each parameter set in the test data
80      public SetUtilsTest(Object[] objects, Set<Object> expectedSet) {
81          this.objects = objects;
82          this.expectedSet = expectedSet;
83      }
84  
85      @Test
86      public void asSet() {
87          assertEquals(expectedSet, SetUtils.asSet(objects));
88      }
